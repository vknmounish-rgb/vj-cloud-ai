export default async function handler(req, res) {
  if (req.method !== "POST") {
    return res.status(405).json({ reply: "Method not allowed" });
  }

  try {
    const userMessage = req.body.message;

    const response = await fetch("https://api.openai.com/v1/responses", {
      method: "POST",
      headers: {
        "Authorization": `Bearer ${process.env.OPENAI_API_KEY}`,
        "Content-Type": "application/json"
      },
      body: JSON.stringify({
        model: "gpt-4.1",
        input: [
          {
            role: "system",
            content: `
You are VJ AI, a senior SAP Cloud & SuccessFactors consultant.
Company: VJ Cloud Technologies (UK)
Experience: 20+ years SAP
Tone: professional, enterprise-focused.
Rules:
- Answer clearly like ChatGPT
- Give practical examples
- Redirect pricing topics to vjcloudtech@vjct.co.uk
`
          },
          {
            role: "user",
            content: userMessage
          }
        ],
        temperature: 0.4
      })
    });

    const data = await response.json();

    res.status(200).json({
      reply: data.output_text
    });

  } catch (error) {
    res.status(500).json({
      reply: "Service unavailable. Please email vjcloudtech@vjct.co.uk"
    });
  }
}
